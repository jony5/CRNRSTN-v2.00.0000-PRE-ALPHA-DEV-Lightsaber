The environmentals :: class object is constructed with its own instantiation of the http_manager :: class object. In the below example of a classic and common client-server interaction, the http_manager :: class object facilitates the handling of data sent to the server via form submission for user authentication.

After determining that $_POST data has indeed been provided (on line 12) through the invocation of issetHTTP(), the extractData() method of the http_manager :: class object is then called to return a parameter ('postid') which both identifies the source of the HTTP request and determines the data handling requirements. This particular request is processed through a web services layer.

<?php
/*
// J5
// Code is Poetry */
require('./_crnrstn.root.inc.php');
require($CRNRSTN_ROOT.'_crnrstn.config.inc.php');    // THE CRNRSTN SUITE :: CONFIGURATION FILE
require($CRNRSTN_ROOT.'_customclass/J5/user.inc.php'); // CUSTOM USER :: CLASS DEFINITION

try{
 //
 // CHECK FOR DATA IN HTTP POST METHOD REQUEST TYPE
 if($oCRNRSTN_ENV->oHTTP_MGR->issetHTTP($_POST)){
  //
  // THERE IS POST DATA TO PROCESS. A SWITCH. ON THE POSTID VARIABLE FROM FORM.
  switch($oCRNRSTN_ENV->oHTTP_MGR->extractData($_POST,'postid')){
   case 'login_main':
    //
    // INITIALIZE USER WITH ENVIRONMENTALS CRNRSTN :: CLASS OBJECT
    if(!isset($oUSER)){
     $oUSER = new user($oCRNRSTN_ENV);
    }
    
    //
    // CHECK FOR PRESENCE OF HTTP $_GET MOBILE INDICATOR
    if($oCRNRSTN_ENV->oHTTP_MGR->issetParam($_GET,'mobile')){
     //
     // SET LANDING PAGE TO MOBILE DEVICE COMPATIBLE DESTINATION
     $oUSER->setLandingPage('/m/');
    }
 
    //
    // PROCESS LOGIN ATTEMPT (WEB SERVICES)
    if($oUSER->login()){
     //
     // WE HAVE A VALID LOGIN ATTEMPT
     echo $oUSER->successMessage;
     
    }else{
     //
     // BAD USERNAME OR PASSWORD IN LOGIN ATTEMPT
     echo $oUSER->errorMessage;
     
    }
   break;
   default:
    //
    // OOPS! WE HAVE POST DATA WITH NO WAY TO PRODESS IT. FIRE ERROR MESSAGE.
    // HOOOSTON...VE HAF PROBLEM!
    throw new Exception('Sorry, I was unable to recognize your request of postid['.$oCRNRSTN_ENV->oHTTP_MGR->extractData($_POST,'postid').'].');
   break;
  }
 }
}catch( Exception $e ) {
   //
   // SEND THIS THROUGH THE LOGGER OBJECT
   self::$oLogger->captureNotice('web_integrations->formHandler(postid)', LOG_EMERG, $e->getMessage());
 
   return true;
  }
?>
