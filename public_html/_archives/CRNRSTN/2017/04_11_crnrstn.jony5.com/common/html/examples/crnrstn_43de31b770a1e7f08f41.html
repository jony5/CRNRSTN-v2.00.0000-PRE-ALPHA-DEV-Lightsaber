<?php
/*

require('./_crnrstn.root.inc.php');
require($CRNRSTN_ROOT.'_crnrstn.config.inc.php');   // THE CRNRSTN SUITE :: CONFIGURATION FILE

//
// IF YOU WANT, IP CAN BE SPECIFIED DIRECTLY IN CODE...BUT DONT DO IT THIS WAY
// AT LARGE SCALE DUE TO MAINENANCE COMPLICATIONS.
if($oCRNRSTN_ENV->oCRNRSTN_IPSECURITY_MGR->denyIPAccess("127.0.0.2")){
 //
 // CLIENT IP MATCHES SPECIFIED IP FOR DENIAL. PROCESS DENIAL HERE
 echo "Sorry. You are denied.";
}else{
 //
 // CLIENT IP DOES NOT MATCH ANY FOR DENIAL. GREEN LIGHT REQUEST HERE.
 echo "Here is the link to that resource which some are not to receive.<br>";
 echo '<a href="resource.html">Link</a>';
 
}

//
// IT IS RECOMMENDED THAT YOU LEVERAGE THE CRNRSTN SUITE CONFIG FUNCTIONALITY TO
// SIMPLIFY THE MANAGEMENT OF DENIAL IP IN YOUR APPLICATION TO ONE PLACE.
// HERE, DENIED_ACCESS_IP HAS BEEN CONFIGURED TO STORE THE DENIAL IP IN THE
// CRNRSTN CONFIG FILE THROUGH defineEnvResource()
if($oCRNRSTN_ENV->oCRNRSTN_IPSECURITY_MGR->denyIPAccess($oCRNRSTN_ENV->getEnvParam('DENIED_ACCESS_IP'))){
 //
 // CLIENT IP MATCHES SPECIFIED IP FOR DENIAL. PROCESS DENIAL HERE

}else{
 //
 // CLIENT IP DOES NOT MATCH ANY FOR DENIAL. GREEN LIGHT REQUEST HERE.
