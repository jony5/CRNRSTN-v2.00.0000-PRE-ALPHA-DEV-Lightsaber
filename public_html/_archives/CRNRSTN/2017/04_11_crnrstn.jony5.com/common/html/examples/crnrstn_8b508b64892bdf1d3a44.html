$cookie_ARRAY = $oENV->oCOOKIE_MGR->getAllCookies();

for($i=0;$i<sizeof($cookie_ARRAY);$i++){
 $cookieCrawler_ARRAY = each($cookie_ARRAY);

 //
 // LIST NAME OF SIMPLE COOKIE
 if(!is_array($cookie_ARRAY[$cookieCrawler_ARRAY['key']]) && isset($cookie_ARRAY[$cookieCrawler_ARRAY['key']])){
  echo 'I found a cookie named :: '.$cookieCrawler_ARRAY['key'].'<br>';
  
  //
  // EXTRACT DATA FROM COOKIE
  echo 'I contain this information :: '.$oENV->oCOOKIE_MGR->getCookie($cookieCrawler_ARRAY['key']);

 }else{

  //
  // WE HAVE AT LEAST ONE 1-DIMENSION ARRAY TO TRAVERSE
  foreach ($cookie_ARRAY[$cookieCrawler_ARRAY['key']] as $tmp_cookie_name_dim1=>$tmp_cookie_value_dim1) {
   //
   // LIST NAME OF 1-DIMENSION ARRAY COOKIE
   if(!is_array($cookie_ARRAY[$cookieCrawler_ARRAY['key']][$tmp_cookie_name_dim1])){
    $tmp_cookie_name_dim1 = $cookieCrawler_ARRAY['key']."[".$tmp_cookie_name_dim1."]";
    echo 'I found a cookie named :: '.$tmp_cookie_name_dim1.'<br>';
    
    //
    // EXTRACT DATA FROM COOKIE
    echo 'I contain this information :: '.$oENV->oCOOKIE_MGR->getCookie($tmp_cookie_name_dim1);

   }
  }
 }
