On line 17 in this example, the navigation state is being pulled from cookie data in order to initialize a parameter for the same within session management.

require('./_crnrstn.root.inc.php');    // PATH...i.e. <?php $ROOT="../"; ? >
require($ROOT.'_crnrstn.config.inc.php');   // CRNRSTN CONFIGURATION FILE

//
// CAN WE GET NAVSTATE INITIALIZATION INFORMATION FROM CLEAR TEXT COOKIE
// THROUGH THE CRNRSTN ENVIRONMENTAL :: CLASS OBJECTS PERSONAL COPY OF
// THE COOKIE_MANAGER :: CLASS OBJECT
$ns_updated = $oENV->oCOOKIE_MGR->getCookie('clientNavigationState');

if($ns_updated!=""){
 //
 // INITIALIZE SESSION FROM COOKIE DATA RETRIEVED THROUGH GETCOOKIE
 $oENV->oSESSION_MGR->setSessionParam('NS', $ns_updated);

}

# NOTE THAT ANY COOKIE_MANAGER METHODS MUST BE INVOKED PRIOR TO OUTPUT
# BEING SENT TO THE CLIENT BROWSER...AS COOKIES ARE PART OF THE HTTP HEADER
?>
